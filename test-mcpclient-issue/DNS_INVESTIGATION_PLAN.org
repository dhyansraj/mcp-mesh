def start_uvicorn_server(self) -> bool:
    """Start uvicorn server in background thread and keep script alive"""

    def run_server():
        """Server function runs in background thread"""
        import uvicorn
        config = uvicorn.Config(
            app=self.fastapi_app,
            host="0.0.0.0",
            port=port,
            log_level="info",
            access_log=True
        )
        server = uvicorn.Server(config)
        server.run()  # Blocks in background thread

    # Start server in background thread
    self.server_thread = threading.Thread(target=run_server, daemon=True)
    self.server_thread.start()

    # Critical: Keep script alive (mimics MCP Mesh behavior)
    self._keep_script_alive()  # ← This blocks main thread

def _keep_script_alive(self):
    """Keep the script alive so server continues running"""
    self._start_heartbeat_process()  # Start background heartbeat

    # Block main thread with server monitoring
    try:
        while True:
            if self.server_thread and self.server_thread.is_alive():
                self.server_thread.join(timeout=1)  # Check every second
            else:
                logger.warning("⚠️ Server thread died, exiting...")
                break
    except KeyboardInterrupt:
        logger.info("🛑 Keyboard interrupt received, shutting down...")

#+BEGIN_SRC python
def _start_heartbeat_process(self):
    """Start background heartbeat process that mimics MCP Mesh registry calls"""

    def heartbeat_worker():
        """Background heartbeat worker that calls registry every 5 seconds"""
        import time, socket, httpx

        # Get host IP (like MCP Mesh does for registry calls)
        try:
            with socket.socket(socket.AF_INET, socket.SOCK_DGRAM) as s:
                s.connect(("8.8.8.8", 80))
                host_ip = s.getsockname()[0]
        except Exception:
            host_ip = "127.0.0.1"

        registry_url = f"http://{host_ip}:8000/heartbeat"
        heartbeat_count = 0

        while True:
            try:
                heartbeat_count += 1
                start_time = time.time()

                # Make HEAD call to registry heartbeat endpoint
                with httpx.Client(timeout=2.0) as client:
                    response = client.head(registry_url)
                    duration = round((time.time() - start_time) * 1000, 1)
                    logger.info(f"💓 Heartbeat #{heartbeat_count} -> {registry_url} ({duration}ms) [status: {response.status_code}]")

                time.sleep(5)  # Every 5 seconds

            except Exception as e:
                duration = round((time.time() - start_time) * 1000, 1)
                logger.info(f"💓 Heartbeat #{heartbeat_count} -> {registry_url} ({duration}ms) [failed: {e}]")
                time.sleep(5)  # Continue despite errors

    # Start heartbeat in daemon thread (doesn't prevent shutdown)
    self.heartbeat_thread = threading.Thread(target=heartbeat_worker, daemon=True)
    self.heartbeat_thread.start()
#+END_SRC

* Phase 5: Fixes and Implementation

** Attempt 1: Event Loop Coordination Fix ✅

*** Root Cause Identified
The DNS resolution issue was caused by competing event loops in MCP Mesh:

1. **Immediate Startup**: `_start_uvicorn_immediately()` in `mesh/decorators.py` started uvicorn with `asyncio.run(server.serve())` during decorator processing
2. **Pipeline Startup**: `StartupOrchestrator` later tried to start its own event loop with `asyncio.run(self._orchestrator.process_once())`
3. **Conflict Result**: "cannot schedule new futures after shutdown" errors when agents tried to make DNS-based MCP calls

This was NOT a DNS resolution problem - it was an event loop lifecycle problem that manifested during DNS calls.

*** Fix Implementation Strategy
**Single Event Loop Coordination**: Modify MCP Mesh to use one coordinated event loop instead of competing ones.

**** 1. Modified `_start_uvicorn_immediately()` (mesh/decorators.py:671)
- **Before**: Created separate event loop with `asyncio.run(server.serve())`
- **After**: Only configure uvicorn server, store for pipeline to start
- **Key Change**: Remove competing `asyncio.run()` call, let pipeline handle server startup

**** 2. Enhanced `StartupOrchestrator` (startup_orchestrator.py)
- **Addition**: Logic to detect "configured" servers that need starting
- **Enhancement**: Start configured servers within pipeline's event loop context
